predictors = ['Age', 'Gender', 'BMI', 'Smoking', 'GeneticRisk', 'PhysicalActivity', 'AlcoholIntake', 'CancerHistory']
target = ['Diagnosis']
In [45]:
cancer_train, cancer_test = train_test_split(cancer, test_size=.3)
In [46]:
clf = LogisticRegression()
clf.fit(cancer_train[predictors], cancer_train[target])
predictions = clf.predict(cancer_test[predictors])
accuracy = metrics.accuracy_score(cancer_test[target], predictions)
cm = metrics.confusion_matrix(cancer_test[target], predictions)
sns.heatmap(cm, annot=True, fmt='.0f')
plt.xlabel('Predicted')
plt.ylabel('Actual')
plt.title('Cancer Diagnosis')
class1_error_rate = cm[1][0]/(cm[1][0]+cm[1][1])
class1_error_rate
class0_error_rate = cm[0][1]/(cm[0][1]+cm[0][0])
class0_error_rate
print('Logistic Regression')
print('Accuracy:', accuracy)
print('Class 1 Error Rate:', class1_error_rate)
print('Class 0 Error Rate:', class0_error_rate)
